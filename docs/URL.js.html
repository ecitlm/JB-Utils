const URL = urlStr => {
  let url = urlStr || window.location.href;
  try {
    let parseUrl = new URL(url);
    parseUrl.params = parseQueryString(url);
    return parseUrl;
  } catch (e) {
    throw new Error('URL无效');
  }
};
export default URL;
